client.Connect();

var cmd = client.CreateCommand("sleep 15s;echo 123"); // Perform long running task 

var asynch = cmd.BeginExecute();

while (!asynch.IsCompleted)
{
    //  Waiting for command to complete...
    Thread.Sleep(2000);
}
result = cmd.EndExecute(asynch);
client.Disconnect();
